 题目链接：http://acm.hdu.edu.cn/status.php?user=XHC91&pid=1011&status=5

这题是一道树形DP，从1号顶点进入树中，每次可以分配小兵到其他可达的顶点去，杀死所有的bugs可以获取brain值，求出m个小兵最多能获取多少brain值。这题要对顶点进行DFS一搜到底，然后用背包的方法选取孩子节点中最多的brain。

对于一个父母顶点，除去它的brain和cost之后，剩下的问题就是一个大小为m-cost的背包了，再由它的孩子节点中任意选取顶点顶点，最终必获得父母顶点的最大值。

```
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define maxn 222
enum boolean {FALSE,TRUE};
int dp[maxn][maxn];
struct room
{
    int bugs;
    int p;
}room[maxn];
int Tree[maxn][maxn];
int NodeCount[maxn];
int Visit[maxn];
int N,M;
int MAX(int x,int y)
{
    return x>y?x:y;
}
int DFS(int p)
{
    int i,j,k;
    int child;
    int mcost;
    Visit[p] = TRUE;
    mcost = (room[p].bugs+19)/20;
    for (i = mcost; i <= M; i++)
        dp[p][i] = room[p].p;
    for (i = 0; i <= NodeCount[p]; i++)
    {
        child = Tree[p][i];
        if (!Visit[child])
        {
            DFS(child);
            for ( j = M; j>=mcost;j--)
                for (k = 1; j+k<= M; k++)  
                    if (dp[child][k])
                        dp[p][j+k] = MAX(dp[p][j+k],dp[p][j]+dp[child][k]);
            }
        }
    }
    return TRUE;
}
int main()
{
    freopen("Sample Input.txt","r",stdin);
    int i,j,k;
    int u,v;
    while (scanf("%d%d",&N,&M),N>=0||M>=0)
    {
        for (i = 1; i <= N; i++)
            scanf("%d%d",&room[i].bugs,&room[i].p);
        memset(NodeCount,-1,sizeof(NodeCount));
        memset(dp,0,sizeof(dp));
        memset(Visit,FALSE,sizeof(Visit));
        for (i = 1; i <= N-1; i++)
        {
            scanf("%d%d",&u,&v);
            Tree[u][++NodeCount[u]] = v;
            Tree[v][++NodeCount[v]] = u;  
        }
        if (!M)
            printf("0\n");
        else
        {
            DFS(1);
            printf("%d\n",dp[1][M]);
        }
    }
    getch();
    return 0;
}
```
