题目链接：http://acm.hdu.edu.cn/showproblem.php?pid=1241

题目解释一下就是找出有多少块有石油的区域，就是数组中的@,这边相邻指的是是周围的八个位置。

算法就是使用DFS对区域进行标记，很水，不多说了。

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int m,n;
int pack;
int find[110][110];
char map[110][110];
int iPlus[8] = {-1,-1,0,-1,0,1,1,1};
int jPlus[8] = {-1,1,-1,0,1,0,-1,1};
void DFS(int i,int j)
{
    int k;
    int near_i,near_j;
    find[i][j] = 1;
    for ( k = 0; k < 8; k++)
    {
        near_i = i + iPlus[k];
        near_j = j + jPlus[k];
        if ( near_i >= 0 && near_i < m && near_j >=0 && near_j < n )
        {
            if(find[near_i][near_j]==0 && map[near_i][near_j]=='@')
                DFS(near_i,near_j);
        }
    }
}
int main()
{
    freopen("Sample Input.txt","r",stdin);
    system("Color F0");
    int i,j,k;
    while ( scanf("%d%d",&m,&n), m&&n)
    {
        for ( i = 0; i < m; i++)
        {
            getchar();
            scanf("%s",map[i]);
        }
        pack = 0;
        memset(find,0,sizeof(find));
        for ( i = 0; i < m; i++)
        {
            for ( j = 0; j < n; j++)
            {
                if ( map[i][j]=='@' && find[i][j]==0)
                {
                    pack++;
                    DFS(i,j);
                }
            }
        }
        printf("%d\n",pack);
    }
    getch();
    return 0;
}
```
